Integration plugin demo advanced

Code:

//plugin init
$('#address-2').addressIntegration({
    countrySelector:                            '#country-2',
    countryShortSelector:                       '#country-short-2',
    citySelector:                               '#city-2',
    stateSelector:                              '#state-2',
    stateShortSelector:                         '#state-short-2',
    postalCodeSelector:                         '#postal-code-2',
    routeSelector:                              '#route-2',
    streetNumberSelector:                       '#street-number-2',
    loaderSelector:                             '#address-integration-loader-2',
    messageSelector:                            '#address-integration-messages-2',
    callbackEventFired:   function() {
        //block button
        $('#check-address').attr('disabled', true);
        //save and show current status of plugin flow
        $('#status').html('<span class="label label-warning">event fired</span>').data('status', 'event fired');
    },
    callbackInProgress:   function() {
        //block button
        $('#check-address').attr('disabled', true);
        //save and show current status of plugin flow
        $('#status').html('<span class="label label-warning">in progress</span>').data('status', 'in progress');
    },
    callbackSuccess:      function(results) {
        //unblock button
        $('#check-address').attr('disabled', false);
        //save and show current status of plugin flow
        $('#status').html('<span class="label label-success">success</span>').data('status', 'success');
        //do some additional things on success
        additionalFunctionalitySuccess(results);
    },
    callbackError:        function(errorMessage) {
        //unblock button
        $('#check-address').attr('disabled', false);
        //save and show current status of plugin flow
        $('#status').html('<span class="label label-danger">error</span>').data('status', 'error');
        //do some additional things on error or when no results
        additionalFunctionalityError(errorMessage);
    }
});

//save and show full results in bootstrap modal when plugin status is success
function additionalFunctionalitySuccess(results) {
    $('#address-results').text($("#address-2").val());
    $('#full-results').text(results[0]['formatted_address']);
    $(".country-result").text($("#country-2").val());
    $(".state-result").text($("#state-2").val());
    $(".city-result").text($("#city-2").val());
    $(".postal-result").text($("#postal-code-2").val());
    $(".street-result").text($("#route-2").val());
    $(".street-number-result").text($("#street-number-2").val());
}

//save and show errors in bootstrap modal when plugin status is fail
function additionalFunctionalityError(errorMessage) {
    $("#address-integration-errors").text(errorMessage);
}

$('#check-address').on('click', function(e) {
    e.preventDefault();
    if($('#status').data('status') === 'success') {
        //show full results in bootstrap modal when plugin status stored in '#status' element is 'success'
        $("#address-integration-results-modal").modal();
    } else {
        //show errors in bootstrap modal when plugin status stored in '#status' element is 'error'
        $("#address-integration-errors-modal").modal();
    }
});
